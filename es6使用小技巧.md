### 1.使用扩展运算符复制数组（浅复制，只对一维数组起作用，相当于slice()）
  let a = [...b]
  
### 2.数组去重(Set类似于数组，但它的值都是唯一的)
  let a = Array.from(new Set([1,1,2,3,4,3,2]))
  a: [1, 2, 3, 4]
  
### 3.解构赋值
  const arr = [1,2,3]
  const [first, second] = arr
 
### 4.字符串去掉最后一个字符
  '1,2,3,'.slice(0,-1)
  "1,2,3"
  
### 5.以一定规则补足字符串
  '1,2,3,'.repeat(3)
  "1,2,3,1,2,3,1,2,3,"
  
  'x'.padStart(3, 'y')
  "yyx"
  
  'x'.padEnd(3, 'y')
  "xyy"
